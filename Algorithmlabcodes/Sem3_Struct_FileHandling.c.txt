#include<stdio.h>
#include<string.h>
#define MAX_STUDENTS 15


struct Student {
    char name[20];
    int roll;
    char telephone[11];
    char address[20];
};
struct SReg {
    int count; 
    struct Student studentArray[MAX_STUDENTS];
};
int add(struct SReg *sr,struct Student s) {




    for(int i = 0; i<sr->count; i++){
        //Checking for duplicate array
        if(sr->studentArray[i].roll == s.roll){
            printf("%d roll number already exists in the student array\n",s.roll);
            return 0;
        }
    }
    // Adding a student
    if(sr->count < MAX_STUDENTS){
        sr->studentArray[sr->count++] = s;
        return 1;
    }
    printf("Size Limit Exceeded\n");
    return 0;


}
struct Student get(struct SReg *sr,int r){


    struct Student NotFound = {0};
    for(int i=0;i<sr->count ;i++){
        if(sr->studentArray[i].roll == r){
            return sr->studentArray[i];
        }
    }
    return NotFound;
}


int delete(struct SReg *sr, int r){


    int index = 0;
    for(int i=0;i<sr->count;i++){
        if(sr->studentArray[i].roll == r){
            index = i;
            break;
        }
    }
    if(index != 0){
        for(int j=index ;j<sr->count;j++){
            sr->studentArray[j] = sr->studentArray[j+1];
        }
        sr->count--;
        printf("The enetered roll got deleted.\n");
        return 1; 
    }
    printf("The roll doesn't exist.\n");
    return 0;
}


int modify(struct SReg *sr,struct Student s){


    for(int i=0;i<sr->count;i++){
        if(sr->studentArray[i].roll == s.roll){
            sr->studentArray[i] = s;
            return 1;
        }
    }
    return 0;
}
void sortStudents(struct SReg *sr){
    for(int i=0;i<sr->count;i++){
        for(int j=0;j<sr->count - 1;j++){
            if(strcmp(sr->studentArray[j].name , sr->studentArray[j+1].name) > 0){
                struct Student temp = sr->studentArray[i];
                sr->studentArray[i] = sr->studentArray[i+1];
                sr->studentArray[i+1] = temp;
            }
        } 
    }
    return ;
}
int getCount(struct SReg *sr){
    return sr->count;
}




void export(struct SReg *sr, char *fname) {
    FILE *fp = fopen(fname, "w");
    if (fp == NULL) {
        printf("Could not open file %s for writing.\n", fname);
        return;
    }


    for (int i = 0; i < sr->count; i++) {
        fprintf(fp, "%s,%d,%s,%s\n",
            sr->studentArray[i].name,
            sr->studentArray[i].roll,
            sr->studentArray[i].telephone,
            sr->studentArray[i].address
        );
    }


    fclose(fp);
    printf("Student data exported to %s successfully.\n", fname);
}




void load(struct SReg *sr,char *fname) {
    FILE *fp = fopen(fname, "r");
    if (fp == NULL) {
        printf("Could not open file %s for reading.\n", fname);
        return;
    }
    struct Student temp;
    sr->count = 0;


    while (fscanf(fp, "%19[^,],%d,%10[^,],%19[^\n]\n",
                  temp.name, &temp.roll, temp.telephone, temp.address) == 4) {
        if (sr->count < MAX_STUDENTS) {
            sr->studentArray[sr->count++] = temp;
        } else {
            printf("Maximum student limit reached while loading.\n");
            break;
        }
    }


    fclose(fp);
    printf("Student data loaded from %s successfully.\n", fname);
}
